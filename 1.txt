5)scripted pipeline
pipline 
write script 

pipeline { 
    agent any

    tools { 
        jdk 'JAVA_HOME'            // configured in Jenkins
        maven 'MAVEN_HOME'         // configured in Jenkins
    } 

    environment {
        // tool returns path like C:\Program Files\... so we assign it to MVN and then always call it quoted
        MVN = "${tool 'MAVEN_HOME'}\\bin\\mvn"
        REPO = "https://github.com/K-Rohit-Reddy/mavenjava.git"
        APP = "mavenjava"
    }

    stages { 

        stage('Git Clone & Clean') { 
            steps { 
                echo "Cleaning old workspace..."
                bat "rmdir /s /q %APP% 2>nul || exit 0"

                echo "Cloning repo..."
                bat "git clone %REPO%"
            } 
        } 

        stage('Maven Clean') { 
            steps { 
                // NOTE the quoted %MVN% to handle spaces in the path
                bat "\"%MVN%\" clean -f %APP%/pom.xml"
            } 
        } 

        stage('Install') { 
            steps { 
                bat "\"%MVN%\" install -f %APP%/pom.xml"
            } 
        } 

        stage('Test') { 
            steps { 
                bat "\"%MVN%\" test -f %APP%/pom.xml"
            } 
        } 

        stage('Package') { 
            steps { 
                bat "\"%MVN%\" package -f %APP%/pom.xml"
            }
        }
    }
}
